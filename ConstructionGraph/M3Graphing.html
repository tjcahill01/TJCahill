fig,ax = plt.subplots()

ax.plot(df_const['Month-Year'],df_const['Private Construction'],color='b')
fig.suptitle('Private & Public Construction Spending by Month', fontsize = 16)

#TimeSeries Data Labels
ax.text(146,46647,'Private',fontsize=13,verticalalignment='center',horizontalalignment='left', color='b')
ax.text(146,16721,'Public',fontsize=13,verticalalignment='center',horizontalalignment='left')

#Formatting the Plot Space
ax.xaxis.set_tick_params(bottom = False)
ax.yaxis.set_tick_params(left = False)
ax.spines['right'].set_visible(False)
ax.spines['top'].set_visible(False)
ax.set_ylim(0,max(df_const['Total Construction'])+50) #Add buffer space 

#Label for y-axis & x-axis
